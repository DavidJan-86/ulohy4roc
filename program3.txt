//Pohyby se pouzitim tlacitka start 2 na DI 10 zopakuji.
//I4 a I0 8 b pro ovládání akcních clenu
//I1 4 vstupní bity DI4, DI5 IR senzory na analogove vstupy AI0 a AI1, DI6 (koncovy spinac zpet)
// a DI7 (koncovy spinac dopredu)
//I2 - 4 b enkodery vertikalni a horizontalni osy
//I3 DI4 koncovy spinac horizontalni osy, DI5 svetelna brana dovnitr, DI6 svetelna brana ven
//a DI7 koncovy spinac vertikalni osy
//O1 a O2 8 b licna, 4 motory/2 smery

//Zde pouzijeme DO8=pohyb0 (dolu), DO9=pohyb1 (nahoru), DI8=start0, DI9=start1, DI7=KVerti,
//DI2 a DI3=IRC vertikalni osa, rozsah vstupu:
//Tlacitko na DI10 (start2) - opakovani.
//vystup DO0 ... sig ... signalizace moznosti opakovani

FUNCTION NaPolohu:BOOL // deklarace funkce
 VAR_INPUT
  polVertik:UDINT; // vstupní promìnné
 END_VAR
 IF hodnotaA2>polVertik AND (hodnotaA2-polVertik)>50 THEN pohyb1:=TRUE;pohyb0:=FALSE;//jedu nahoru
  ELSIF hodnotaA2<polVertik AND (polVertik-hodnotaA2)>50 THEN pohyb0:=TRUE;pohyb1:=FALSE;//dolu
  ELSE pohyb0:=FALSE; pohyb1:=FALSE;
 END_IF;
 IF (NOT pohyb0 AND NOT pohyb1) = TRUE THEN
  NaPolohu:=TRUE;
   ELSE NaPolohu:=FALSE;
 END_IF;
//jsme na zadane poloze... motory vypináme chvili pred dosazenim zadane polohy
END_FUNCTION


PROGRAM prgMain
        VAR
           cisloA2:UDINT;
	         KS2:BOOL;
	         Start:BOOL:=TRUE;
	         Pom0,Pom1,Pom2,Pom3,Pom4,Pom5,Pom6,Pom7,Pom8,Pom9,Pom10:BOOL;
	         Vys0,Vys1,Vys2,Vys3,Vys4,Vys5,Vys6,Vys7,Vys8,Vys9,Vys10:BOOL;
	         NaPolohu0,NaPolohu1,NaPolohu2,NaPolohu3:BOOL;
	         polVertik0,polVertik1,polVertik2,polVertik3:UDINT;
	         Casovac0,Casovac1,Casovac2,Casovac3,Casovac4:TON;
	         Cas0,Cas1,Cas2,Cas3,Cas4:TIME;
	         NabHrPom0,NabHrVys1,NabHrVys3,NabHrVys5,NabHrVys7:R_TRIG;
	         max:UDINT:=4200;//dolni krajni poloha
        END_VAR
//enkodery
//zadat u CNT_OUT2 v poli CCNT ridiciSlovo2 (UINT) a u CNT_IN2 v poli VALA hodnotaA2 (UDINT),
//zadat IRC u IB1301 - bity 2 a 3
        cisloA2:=hodnotaA2;//nahoru/dolu ... bity 2 a 3 IB1301
//koncove spinace
        KS2:=KVerti;//koncovy spinac vertikalni osy
//akcni cleny
        pohyb0:=start0;//dolu
        pohyb1:=start1;//nahoru
IF Start=TRUE THEN
  IF KS2=FALSE THEN pohyb1:=TRUE;//rameno nahoru
   ELSE pohyb1:=FALSE;Start:=FALSE;Pom0:=TRUE;
  END_IF;
END_IF;
NabHrPom0(CLK:=Pom0,Q=>Vys0);
IF Vys0=TRUE THEN
	ridiciSlovo2:=2;
	Pom1:=TRUE;
	Pom0:=FALSE;
END_IF;
Casovac0(IN:=Pom1,PT:=T#3s,Q=>Vys1,ET=>Cas0);
NabHrVys1(CLK:=Vys1,Q=>Vys2);
IF Vys2=TRUE THEN
	ridiciSlovo2:=1;
	polVertik0:=max;
	Pom1:=FALSE;
	Pom2:=TRUE;
END_IF;	

IF Pom2=TRUE THEN
	NaPolohu0:=NaPolohu(polVertik0);
	IF NaPolohu0=TRUE THEN
		Pom2:=FALSE;
		Pom3:=TRUE;
	END_IF;
END_IF;

Casovac1(IN:=Pom3,PT:=T#5s,Q=>Vys3,ET=>Cas1);
NabHrVys3(CLK:=Vys3,Q=>Vys4);
IF Vys4=TRUE THEN
	polVertik1:=max/4;
	Pom3:=FALSE;
	Pom4:=TRUE;
END_IF;	
IF Pom4=TRUE THEN
	NaPolohu1:=NaPolohu(polVertik1);
	IF NaPolohu1=TRUE THEN
		Pom4:=FALSE;
		Pom5:=TRUE;
	END_IF;
END_IF;
Casovac2(IN:=Pom5,PT:=T#2s,Q=>Vys5,ET=>Cas2);
NabHrVys5(CLK:=Vys5,Q=>Vys6);
IF Vys6=TRUE THEN
	polVertik2:=max*3/4;
	Pom5:=FALSE;
	Pom6:=TRUE;
END_IF;	
IF Pom6=TRUE THEN
	NaPolohu2:=NaPolohu(polVertik2);
	IF NaPolohu2=TRUE THEN
		Pom6:=FALSE;
		Pom7:=TRUE;
	END_IF;
END_IF;
Casovac3(IN:=Pom7,PT:=T#7s,Q=>Vys7,ET=>Cas3);
NabHrVys7(CLK:=Vys7,Q=>Vys8);
IF Vys8=TRUE THEN
	polVertik3:=max/2;
	Pom7:=FALSE;
	Pom8:=TRUE;
END_IF;	
IF Pom8=TRUE THEN
	NaPolohu3:=NaPolohu(polVertik3);
	IF NaPolohu3=TRUE THEN
		Pom8:=FALSE;
		Pom9:=TRUE;
	END_IF;
END_IF;

IF Pom9=TRUE THEN

	  Pom9:=FALSE;
	  Start:=TRUE;
 	  sig:=FALSE;
	END_IF;
END_IF;
END_PROGRAM
